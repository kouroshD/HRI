94 void AOnode::isFeasible()
95 {
96  bool temp_isFeasible = false;
97 
98  // 1. the node is feasible if it is already feasible
99  if (nFeasible == true)
100  temp_isFeasible = true;
101 
102  // 2. the node is feasible if it is a terminal node (no hyperarcs)
103  if (arcs.size() == 0)
104  temp_isFeasible = true;
105 
106  // 3. the node is feasible if it has >=1 hyperarcs with all child nodes solved
107  // iterate on the hyperarcs of the node
108  for (int i=0; i<(int)arcs.size(); i++)
109  {
110  bool allSolved = true;
111 
112  // iterate on the child nodes
113  for (int j=0; j<(int)arcs[i].children.size(); j++)
114  {
115  // if the node is not solved --> break
116  if (arcs[i].children[j]->nSolved == false)
117  {
118  allSolved = false;
119  break;
120  }
121  }
122  // if all the child nodes are solved, this node is feasible
123  if (allSolved == true)
124  {
125  temp_isFeasible = true;
126  break;
127  }
128  }
129 
130  // set the feasibility property
131  nFeasible = temp_isFeasible;
132 }
